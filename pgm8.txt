#include <stdio.h>
#define MAX 10
int x[MAX], s[MAX]; 
int n, d, count = 0; 

void sumofsub(int sum, int k, int r) {
    int i;

    x[k] = 1;
    if (sum + s[k] == d) {
        printf("\nSubset %d: ", ++count);
        for (i = 0; i <= k; i++)
            if (x[i] == 1)
                printf("%d ", s[i]);
    } else if ((sum + s[k] + s[k + 1]) <= d) {
        sumofsub(sum + s[k], k + 1, r - s[k]); 
    }

  
    if ((sum + r - s[k]) >= d && (sum + s[k + 1]) <= d) {
        x[k] = 0;
        sumofsub(sum, k + 1, r - s[k]);
    }
}

int main() {
    int i, sum = 0;

    printf("Enter the n value: ");
    scanf("%d", &n);

    printf("Enter the set in increasing order:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &s[i]);
        sum += s[i];
    }

    printf("Enter the max subset value: ");
    scanf("%d", &d);

    if (sum < d) {
        printf("\nNo subset possible\n");
    } else {
        printf("\nSubsets with sum %d are:\n", d);
        sumofsub(0, 0, sum);
    }

    return 0;
}
